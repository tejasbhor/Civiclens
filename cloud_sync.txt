# =====================================================
# CivicLens — Cloud Sync & Redeploy Script
# Run from: /opt/civiclens
# =====================================================

# 1. Pull latest code
git pull origin main

# 2. ALWAYS force ENABLE_DEMO_OTP=true (critical for demo OTP visibility)
# sed replaces it if exists; fallback appends if not
if grep -q "ENABLE_DEMO_OTP" .env; then
    sed -i 's/^ENABLE_DEMO_OTP=.*/ENABLE_DEMO_OTP=true/' .env
else
    echo "ENABLE_DEMO_OTP=true" >> .env
fi
echo "✅ ENABLE_DEMO_OTP confirmed = $(grep ENABLE_DEMO_OTP .env)"

# 3. Stop all containers (volumes are preserved — no data loss)
sudo docker compose down

# 4. Rebuild sequentially to conserve RAM on OCI free-tier
echo "🏗️ [1/3] Rebuilding Backend..."
sudo docker compose build backend

echo "🏗️ [2/3] Rebuilding Admin Dashboard (memory-limited)..."
NODE_OPTIONS="--max-old-space-size=1024" sudo docker compose build admin

echo "🏗️ [3/3] Rebuilding Citizen Portal (no-cache forced)..."
sudo docker compose build --no-cache client

# 5. Start all services
sudo docker compose up -d --remove-orphans

# 6. Apply database migrations
echo "⌛ Waiting for DB to stabilize..."
sleep 10
sudo docker compose exec -T backend alembic upgrade head

echo ""
echo "✅ ALL DONE! Services are live:"
echo "   → Citizen Portal : https://civiclens.space/citizen/login"
echo "   → Officer Portal  : https://civiclens.space/officer/login"
echo "   → API Health      : https://api.civiclens.space/health"
